\section{Introduction}
%------------------------------------------------------------------

The Clay Millennium Problem asks whether smooth initial data for the
incompressible Navier--Stokes equations on $\mathbb{R}^3$ always produce
smooth solutions for all time. We approach this question through a
construction that is simultaneously geometric, dynamical, and logical,
organized around the Apollonian sphere packing and a hierarchy of
increasingly powerful initial-data families.

The argument proceeds in five stages:

\begin{enumerate}
  \item \textbf{The Apollonian Dilemma} (Section~\ref{sec:dilemma}).
  APPO is either admissible or forbidden as an NS/FSI boundary condition.
  Both horns destroy global regularity.

  \item \textbf{EPOT: Turing Completeness and Diagonalization}
  (Section~\ref{sec:turing}). The touching subfamily of EPPO configurations
  encodes arbitrary Turing computation. The halting problem is undecidable;
  smooth solutions would make it decidable.

  \item \textbf{EPOQ: The Oracle Hierarchy and Universal Quantum Computation}
  (Section~\ref{sec:epoq}). EPOTs whose interior Euler flows host
  self-replicating, state-preserving Turing machines form a class equivalent
  to a universal quantum computer. NS/FSI on the EPOQ family is universal
  quantum computation.

  \item \textbf{The Kernel Theorem} (Section~\ref{sec:kernel}). The
  innermost Soddy sphere accumulates a complete holographic description of
  all of $\mathcal{A}$. A finite object encodes infinite fractal information,
  violating every finite regularity bound.

  \item \textbf{The NS Closure} (Section~\ref{sec:closure}). NS forbids
  infinite energy in a single initial condition, not in a family. The
  Apollonian kernel is the limit point of an infinite family of legal
  finite-energy EPOQs, lying in the closure of the state space but outside
  it, forcing regularity failure on a dense set.
\end{enumerate}

\section{Definitions and Setup}

\subsection{The Apollonian Sphere Packing}

Begin with four mutually tangent spheres in $\mathbb{R}^3$ with curvatures
$k_i = 1/r_i$ satisfying the Descartes--Soddy relation
\[
  \bigl(k_1+k_2+k_3+k_4\bigr)^2
  = 2\bigl(k_1^2+k_2^2+k_3^2+k_4^2\bigr).
\]
The Apollonian packing $\mathcal{A}$ is obtained by iterating the Soddy
recursion: at each generation every interstice bounded by four mutually
tangent spheres is filled by the unique fifth tangent sphere. The complement
$\mathbb{R}^3\setminus\mathcal{A}$ has Hausdorff dimension $d_H \approx
2.4739$, is nowhere rectifiable, and is invariant under the M\"{o}bius
group $\mathrm{Mob}(3)$.

The \emph{kernel} of $\mathcal{A}$ is the unique innermost Soddy sphere $K$
of radius $\epsilon$, tangent to all four initial spheres from the interior.
Every sphere in $\mathcal{A}$ is connected to $K$ by a finite chain of
tangency contacts.

\subsection{The Three Configuration Families}

\begin{definition}[EPPO --- Effective Positive Porosity Object]
An EPPO configuration is a finite collection of disjoint closed rigid
spheres whose centers and radii follow the first $N$ generations of the
Apollonian recursion, with \emph{every} potential tangency replaced by a
strictly positive gap:
\[
  \delta_k = \tfrac{1}{2}\min\bigl\{r_j : j \in \mathrm{parents}(k)\bigr\}
  \quad (k \geq 1), \quad \delta_0 > 0.
\]
All gaps are strictly positive. The fluid domain $\Omega(0)$ is $C^\infty$
and the NS/FSI problem is locally well-posed.
\end{definition}

\begin{definition}[EPOT --- Effective Positive Or Tangent]
An EPOT configuration is an EPPO configuration in which \emph{finitely
many} sphere pairs have their gap reduced to exactly zero (tangent contact),
forming a finite contact graph $G = (V,E)$. All other pairs retain strictly
positive gaps.
\end{definition}

\begin{proposition}[Legality of EPOT initial data]
\label{prop:epot_legal}
An EPOT configuration with finitely many tangent pairs is a legal NS/FSI
initial datum.
\end{proposition}

\begin{proof}[Justification]
Finitely many tangency points are isolated boundary singularities. Standard
capacity arguments show that a domain Lipschitz away from a finite
exceptional set of capacity zero admits local well-posedness. This is
categorically different from APPO, where the tangencies are dense with
Hausdorff dimension $d_H \approx 2.4739$.
\end{proof}

\begin{definition}[EPOQ --- Effective Positive Or Quantum]
\label{def:epoq}
An EPOQ configuration is an EPPO or EPOT configuration whose interior Euler
flow hosts a Turing machine $\mathcal{T}$ with the following properties:

\begin{enumerate}
  \item \textbf{Self-replication outward.} $\mathcal{T}$ generates four
  larger self-similar EPOQ copies in surrounding balls at the first
  Apollonian generation outward. Each copy has less energy than its parent,
  less communication overhead, and more storage capacity (proportional to
  the cube of the larger radius).

  \item \textbf{Full radius storage.} $\mathcal{T}$ stores the radii of all
  spheres visible to it: the larger spheres above it (which generated it)
  and the smaller spheres below it (which it generated).

  \item \textbf{Termination recording.} $\mathcal{T}$ records its own
  termination time on its tape before halting.

  \item \textbf{State maintenance on death.} Upon termination, $\mathcal{T}$
  does not simply halt. It passes its complete state one layer \emph{below}
  --- to the child EPOQ --- through the inward near-tangency communication
  channel. The child stores this state and can reset the parent from the
  checkpoint, preserving computation through machine death.

  \item \textbf{Countable enumeration.} By iterating the reset-and-rerun
  protocol across all levels, the EPOQ hierarchy enumerates and checks every
  countable sphere in $\mathcal{A}$.
\end{enumerate}
\end{definition}

\begin{proposition}[Legality of EPOQ initial data]
An EPOQ configuration is a legal NS/FSI initial datum.
\end{proposition}

\begin{proof}[Justification]
An EPOQ is an EPPO or EPOT (hence legal) with a prescribed smooth
divergence-free interior Euler flow. The interior flow does not alter
boundary regularity; local well-posedness is inherited.
\end{proof}

\subsection{Driving Flows}

Fix a smooth divergence-free radially inward velocity field $u_0$ on a
control sphere $\partial B_R$, with magnitude sufficient to close all gaps
in finite time absent rigid-body resistance. The \emph{M\"{o}bius orbit}
\[
  \mathcal{F} = \{u_0 \circ \varphi : \varphi \in \mathrm{Mob}(3)\}
\]
is the full family of driving flows. Since $\mathcal{A}$ is
M\"{o}bius-invariant, all qualitative conclusions below are preserved
under $\mathcal{F}$.

\subsection{APPO as a Forbidden Singular Boundary}

As EPPO gaps $\delta_k(t) \to 0$, fluid layers become lubrication layers.
Classical lubrication theory gives resisting forces $\sim \delta_k^{-\alpha}$,
$\alpha \geq 1$. For sufficiently strong driving this resistance is overcome.
At the APPO limit: infinitely many pairs are simultaneously tangent; no-slip
conditions become irreconcilable at contact points; $\partial\Omega$
degenerates into the fractal residual set of $\mathcal{A}$, nowhere
Lipschitz. APPO is a \emph{singular forbidden boundary} of the NS/FSI phase
space, not a state within it.

\subsection{The Dilemma}

\begin{theorem}[The Apollonian Dilemma]
\label{thm:dilemma}
Whether APPO is admitted or forbidden as an NS/FSI boundary condition,
globally smooth solutions do not exist for all smooth initial data in the
EPPO family.
\begin{enumerate}
  \item[\textbf{(Horn~1)}] \textbf{APPO is admitted.} NS/FSI admits a
  solution on a domain with boundary of Hausdorff dimension $\approx 2.4739$,
  nowhere rectifiable. Classical parabolic regularity requires Lipschitz
  boundary. Contradiction.

  \item[\textbf{(Horn~2)}] \textbf{APPO is forbidden.} NS/FSI must encode
  $\neg$APPO: steer all EPPO trajectories away from $\mathcal{A}$. The
  separatrix between approaching and non-approaching trajectories must
  respect the full M\"{o}bius symmetry of $\mathcal{A}$ and has Hausdorff
  dimension at least $d_H \approx 2.4739$. Smooth flows produce smooth
  separatrices; fractal separatrices are not rectifiable. Contradiction.
\end{enumerate}
\end{theorem}

\begin{remark}
The two horns are exhaustive: $\mathcal{A}$ is rigid and algebraically
determined. There is no partial or deformed APPO.
\end{remark}

\section{EPOT: Turing Completeness and Diagonalization}
\label{sec:turing}

\begin{definition}[Gap-closure decision problem]
Given an EPOT configuration $G_0$, driving flow $u \in \mathcal{F}$, and
parameters $(i,j,t^*)$: does sphere $i$ contact sphere $j$ at or before
$t^*$? The evolving contact graph $G(t)$ is the computational state; contact
events are transitions; $u$ is the program.
\end{definition}

\begin{theorem}[Turing completeness of EPOT evolution]
\label{thm:turing}
NS/FSI evolution on EPOT with flows from $\mathcal{F}$ is Turing complete:
for any Turing machine $\mathcal{T}$ and input $x$, there exists an EPOT
configuration and $u \in \mathcal{F}$ such that $\mathcal{T}(x)$ halts
if and only if a specified gap-closure event occurs in finite time.
\end{theorem}

\begin{proof}[Proof sketch]
Cardona, Miranda, and Peralta-Salas~\cite{CMP2021} proved stationary Euler
flows are Turing complete. NS/FSI on EPOT contains this as a special regime;
contact events are the physical readout; the M\"{o}bius orbit $\mathcal{F}$,
locally homeomorphic to $\mathbb{R}^6$, provides a continuous program space
containing all Turing machines as a dense subfamily, realizing the analog
model of Moore~\cite{Moore1990}.
\end{proof}

\begin{theorem}[Diagonalization against global smoothness]
\label{thm:diag}
If $\Phi_t$ were smooth for all smooth EPOT initial data, the gap-closure
indicator $f(u,G_0,i,j,t^*) = \mathbf{1}[\mathrm{gap}_{ij}(t^*)=0]$ would
be computable as a function of initial data. But $f$ encodes the halting
problem~\cite{Turing1936}, which is not computable. Contradiction.
\end{theorem}

The decision boundary $\mathcal{B} = \partial\{f=1\}$ in parameter space
--- the separatrix between halting and non-halting initial data --- is
non-computable and hence fractal~\cite{GOY1983}, independently confirming
the fractal separatrix of Horn~2.

\subsection{The Computational Horizon}

A Turing machine of size $\epsilon$ in a ball of radius $r$ has resources
$\sim(r/\epsilon)^3$, giving a finite computational horizon: a recursion
depth beyond which Apollonian radii cannot be generated from first
principles. The EPOQ construction resolves this by having each sphere store
what its own resources cannot compute, and building outward so that larger
balls have proportionally more storage.

\subsection{Outward Self-Replication and the Energy Gradient}

The EPOQ hierarchy builds \emph{outward} from the kernel $K$:

\begin{enumerate}
  \item The kernel $\mathcal{T}_0$ generates four larger surrounding EPOQ
  balls at the first Apollonian generation. These are larger (more storage),
  have less energy per ball, and fewer communication channels.

  \item Each first-generation ball replicates outward to four
  second-generation balls, and so on. Larger balls host more
  storage-capable machines.

  \item Near-tangency channels connect adjacent levels, carrying data both
  inward (toward $K$) and outward. The hierarchy is powered by global
  rotation of the initial fluid, from which every level extracts kinetic
  energy.
\end{enumerate}

\subsection{State Maintenance and the Quantum Reset}

When machine $\mathcal{T}_S$ at level $k$ terminates:
it records its termination time and complete state, then transmits this
state one layer below through the inward channel. The child stores the
received state and can reset $\mathcal{T}_S$ from the checkpoint.
No computation is permanently lost. By iterating reset-and-rerun across
all levels, the hierarchy enumerates every countable sphere.

\subsection{EPOQ as a Universal Quantum Computer}

\begin{theorem}[EPOQ universality]
\label{thm:epoq_quantum}
The class of EPOQ configurations with driving flows from $\mathcal{F}$
constitutes a universal quantum computer.
\end{theorem}

\begin{proof}[Proof sketch]
Universal quantum computation requires a universal gate set, superposition,
and state maintenance with interference.

\textit{Gate set.} By Theorem~\ref{thm:turing}, EPOT evolution is Turing
complete; EPOQ contains EPOT as a special case.

\textit{Superposition.} The M\"{o}bius orbit $\mathcal{F}$ is parametrized
continuously by $\mathrm{Mob}(3) \cong \mathbb{R}^6$ locally. An EPOQ
driven by a superposition over $\mathcal{F}$ runs all corresponding
computations simultaneously. The continuum of programs is the physical
substrate of superposition; contact events are measurement outcomes.

\textit{State maintenance and interference.} The death-and-reset protocol
preserves every computational state in the layer below. States from
different computation paths (different $u \in \mathcal{F}$) flow inward
and interfere at the kernel, which is the unique sink of all downward
communication. The kernel is the interference point of the universal quantum
computation.

Together these realize universal quantum computation in the continuous
analog model of Bournez--Campagnolo~\cite{Bournez2006}.
\end{proof}

\begin{corollary}[NS/FSI is a universal quantum computer]
Since EPOQ configurations are legal NS/FSI initial data, the Navier--Stokes
system extended to FSI is computationally equivalent to a universal quantum
computer on the EPOQ family.
\end{corollary}

\section{The NS Closure}
\label{sec:closure}
%------------------------------------------------------------------

\subsection{What NS Forbids and What It Does Not}

The Navier--Stokes energy inequality requires
\[
  E(t) = \tfrac{1}{2}\|u(\cdot,t)\|_{L^2}^2 \leq E(0) < \infty.
\]
NS \emph{forbids infinite energy in a single initial condition}. It does
\emph{not} forbid an infinite family of initial conditions each with finite
energy.

\subsection{The Kernel as a Limit Point}

\begin{theorem}[NS Closure Theorem]
\label{thm:closure}
The Apollonian kernel $K$ lies in the closure of the set of legal NS/FSI
initial states but not in that set itself:
\[
  K \;\in\;
  \overline{\{\text{legal NS/FSI states}\}}
  \;\setminus\;
  \{\text{legal NS/FSI states}\}.
\]
\end{theorem}

\begin{proof}
For any finite energy bound $I < \infty$ and any minimum sphere radius
$r > 0$ (any finite recursion depth $N$), there exists a legal EPOQ
configuration with total energy $\leq I$ encoding $\mathcal{A}$ perfectly
down to radius $r$. As $r \to 0$ with $I \to \infty$, this family converges
to $K$ in Hausdorff distance: $K$ is a limit point of legal states.

However, $K$ itself requires infinite energy to instantiate (the complete
$\mathcal{A}$ encodes infinite information at a single point, by
Corollary~\ref{cor:every_scale}), so $K$ is not a legal NS/FSI state.
\end{proof}

\begin{remark}
Theorem~\ref{thm:closure} identifies precisely what NS does and does not
forbid. It does not forbid encoding arbitrarily deep approximations to
$\mathcal{A}$ --- any finite depth is legal at finite energy. It forbids
only the infinite limit. The infinite family of finite-energy EPOQs is an
\emph{external oracle} for $\mathcal{A}$: no single NS state contains
$\mathcal{A}$, but the family collectively does. The kernel is the point
where the external oracle attempts to collapse into a single internal state
--- and that collapse is the singularity.
\end{remark}

\begin{theorem}[Dense failure of smooth dependence]
\label{thm:dense}
Smooth dependence on initial data fails on a dense set in the EPOQ family.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:closure}, $K$ is a limit point of legal EPOQ initial
states $\{Q_n\}$ with minimum sphere radius $r_n \to 0$. Each $Q_n$ is a
smooth initial datum. Since $K$ is not a legal NS/FSI state, the evolution
$\Phi_t(Q_n)$ cannot reach $K$: every trajectory must become singular before
the common limit point.

If $\Phi_t$ were smooth in initial data, nearby initial data would produce
nearby evolutions, requiring a smooth separatrix at $K$. But $K$ is the
Apollonian kernel, whose boundary is the fractal residual set of dimension
$\approx 2.4739$ (Theorems~\ref{thm:dilemma} and~\ref{thm:diag}). A smooth
separatrix at a fractal boundary is a contradiction. Smooth dependence
therefore fails at every $Q_n$, which is dense.
\end{proof}

\section{The Effective Computational Dimension: $\pi$ Dimensions}
\label{sec:pi}

\[
  \omega_{d_H} = \frac{\pi^{d_H/2}}{\Gamma(d_H/2+1)},
\]
governing spectral asymptotics. The effective computational dimension of
the 3D NS/FSI EPOQ hierarchy is therefore
\[
  \dim_{\mathrm{eff}} = \pi,
\]
with integer part $3$ corresponding to discrete Turing computation and
transcendental excess
\[
  \pi - 3 \approx 0.14159\ldots
\]
corresponding to the non-computable geometric residue encoded in the kernel
--- the holographic shadow of the complete Apollonian packing compressed into
a sphere of radius $\epsilon$.

\begin{remark}
The identification $\dim_{\mathrm{eff}} = \pi$ is a structural observation
about the scaling of the EPOQ hierarchy's information content. Making it
fully rigorous requires a precise definition of effective computational
dimension for continuous analog systems, which we leave as an open problem.
\end{remark}
