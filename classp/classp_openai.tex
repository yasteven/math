\begin{abstract}
The Fact of Fractal Tennis (TFOFT) establishes the ontology: the universe is fundamentally a finite operational computational object defined by a dynamical Apollonian--Soddy Sphere Packing (ASSP) fractal quine. In that ontology, continuum field theories are not fundamental. They are bounded-resolution effective descriptions of finite tangency updates. This paper develops the first continuum-limit equations of the ASSP program. We define an ASSP tangency graph whose vertices are spheres, whose edges are tangencies, and whose primitive local clock variable is the sphere radius. Writing
\[
  u_i = \ln\frac{R_i}{R_0},
\]
we show that the radius-clock equation
\[
  \Delta_G u_i = \frac{4\pi G}{c^2}\rho_i
\]
coarse-grains to Newtonian gravity and to the weak-field time component of general relativity through
\[
  \Phi = c^2 u, \qquad g_{00}=-c^2e^{2u}.
\]
We then construct an Ord-type directed tangency walk on the same graph. Light-speed propagation along oriented tangencies plus a local Gyro-Ball chirality-mixing angle
\[
  \theta_i = \frac{mc^2\Delta\tau_i}{\hbar}
\]
coarse-grains to the Dirac equation. Thus general relativity and quantum mechanics arise as opposite continuum shadows of the same finite ASSP update: radius-clock gradients become gravity, while directed tangency zigzags become quantum matter. Finally, we translate the original TFOFT $3d_{z^2}$ Milky Way prediction into a simulation and observational program: spiral galaxies should be modeled with dark/inertial halo fields following hydrogenic orbital boundary geometries, and the Milky Way should contain a lower-contrast equatorial ``Fermi donut'' corresponding to the toroidal lobe of the same $3d_{z^2}$ state whose polar lobes appear as the Fermi bubbles.
\end{abstract}

\section{General Relativity as the Radius-Clock Continuum Limit}

\subsection{Normalized Graph Laplacian}

Let $f_i=f(x_i)$ be a slowly varying field on the tangency graph. Define a weighted graph Laplacian
\begin{equation}
  (\Delta_G f)_i
  = A_i\sum_{j\sim i}w_{ij}(f_j-f_i),
  \label{eq:lap_general}
\end{equation}
where $w_{ij}=w_{ji}>0$ is a tangency conductance. A natural radius-symmetric choice is
\begin{equation}
  w_{ij}=\frac{4\pi}{R_i+R_j}
  \left(\frac{R_iR_j}{R_i+R_j}\right)^2,
  \label{eq:weights}
\end{equation}
though the continuum proof requires only local symmetry and finite second moments.

Define the weighted second moment tensor
\begin{equation}
  M_i^{ab}=\sum_{j\sim i}w_{ij}s_{ij}^as_{ij}^b,
  \qquad
  M_i=\sum_{j\sim i}w_{ij}\abs{s_{ij}}^2.
\end{equation}
Assume local coarse-grained balance and isotropy:
\begin{align}
  \sum_{j\sim i}w_{ij}s_{ij}^a &=0, \label{eq:balance}\\
  M_i^{ab} &= \frac{M_i}{3}\delta^{ab}. \label{eq:isotropy}
\end{align}
Choose
\begin{equation}
  A_i=\frac{6}{M_i}.
  \label{eq:norm}
\end{equation}
Then
\begin{equation}
  (\Delta_G f)_i
  =
  \frac{6}{\sum_{j\sim i}w_{ij}\abs{s_{ij}}^2}
  \sum_{j\sim i}w_{ij}(f_j-f_i).
  \label{eq:normalized_lap}
\end{equation}

\subsection{Coarse-Graining Proof}

Taylor-expand $f_j=f(x_i+s_{ij})$:
\begin{equation}
  f_j-f_i
  =s_{ij}^a\partial_a f
  +\frac12s_{ij}^as_{ij}^b\partial_a\partial_b f
  +O(\abs{s_{ij}}^3\nabla^3f).
\end{equation}
Substitute into Eq.~\eqref{eq:lap_general}. The first-order term vanishes by Eq.~\eqref{eq:balance}. The leading term is
\begin{align}
  (\Delta_G f)_i
  &=\frac{A_i}{2}
  \sum_{j\sim i}w_{ij}s_{ij}^as_{ij}^b\partial_a\partial_b f
  +O(h_i\nabla^3f)\\
  &=\frac{A_i}{2}\frac{M_i}{3}\delta^{ab}\partial_a\partial_b f
  +O(h_i\nabla^3f).
\end{align}
Using $A_i=6/M_i$ gives
\begin{equation}
  (\Delta_G f)_i = \nabla^2 f + O(h_i\nabla^3 f).
  \label{eq:lap_limit}
\end{equation}
Thus, in the slowly varying locally isotropic limit,
\begin{equation}
  \Delta_G \longrightarrow \nabla^2.
\end{equation}

\subsection{Poisson Gravity}

Postulate the ASSP radius-clock field equation
\begin{equation}
  \Delta_Gu_i = \frac{4\pi G}{c^2}\rho_i.
  \label{eq:discrete_poisson}
\end{equation}
Using Eq.~\eqref{eq:lap_limit}, the continuum limit is
\begin{equation}
  \nabla^2u=\frac{4\pi G}{c^2}\rho.
\end{equation}
Multiplying by $c^2$ and using $\Phi=c^2u$ gives
\begin{equation}
  \boxed{\nabla^2\Phi=4\pi G\rho.}
  \label{eq:poisson}
\end{equation}
Therefore Newtonian gravity is the smooth limit of the ASSP radius-clock equation.

The corresponding slow-particle acceleration follows from Eq.~\eqref{eq:metric}. Since
\[
  g_{00}=-c^2e^{2u},
\]
the weak static geodesic acceleration is
\begin{equation}
  \vec a
  =-\frac{c^2}{2}\nabla\ln\left(-\frac{g_{00}}{c^2}\right)
  =-\frac{c^2}{2}\nabla(2u)
  =-c^2\nabla u
  =-\nabla\Phi.
  \label{eq:acceleration}
\end{equation}
For a spherical mass $M$,
\[
  \Phi(r)=-\frac{GM}{r},
  \qquad
  u(r)=-\frac{GM}{c^2r},
\]
and hence
\begin{equation}
  R(r)=R_0\exp\left(-\frac{GM}{c^2r}\right)
  \simeq R_0\left(1-\frac{GM}{c^2r}\right),
\end{equation}
so local radius, local clock rate, and local coordinate light speed decrease in a gravitational potential well.

\section{Quantum Mechanics as the Directed-Tangency Continuum Limit}

\subsection{One-Dimensional Ord Walk}

The minimal Ord-type limit uses two directed amplitudes,
\[
  \psi_R(x,t), \qquad \psi_L(x,t),
\]
representing right- and left-moving tangency states. Let the local step length be $\epsilon$ and the local proper tick be
\begin{equation}
  \Delta\tau=\frac{\epsilon}{c}.
\end{equation}
Massless propagation is one edge per light-clock tick. Mass enters as a local Gyro-Ball chirality-mixing angle:
\begin{equation}
  \theta=\frac{mc^2\Delta\tau}{\hbar}.
  \label{eq:theta}
\end{equation}
The discrete update is
\begin{align}
  \psi_R(x+\epsilon,t+\Delta\tau)
  &=\cos\theta\,\psi_R(x,t)-i\sin\theta\,\psi_L(x,t),\label{eq:walk_R}\\
  \psi_L(x-\epsilon,t+\Delta\tau)
  &=\cos\theta\,\psi_L(x,t)-i\sin\theta\,\psi_R(x,t).\label{eq:walk_L}
\end{align}
For $\theta\ll1$ and slowly varying amplitudes,
\begin{align}
  \Delta\tau\,\partial_t\psi_R+\epsilon\,\partial_x\psi_R
  &=-i\theta\psi_L,\\
  \Delta\tau\,\partial_t\psi_L-\epsilon\,\partial_x\psi_L
  &=-i\theta\psi_R.
\end{align}
Using $\epsilon/\Delta\tau=c$ and $\theta/\Delta\tau=mc^2/\hbar$ gives
\begin{align}
  (\partial_t+c\partial_x)\psi_R
  &=-i\frac{mc^2}{\hbar}\psi_L,\label{eq:chiral_R}\\
  (\partial_t-c\partial_x)\psi_L
  &=-i\frac{mc^2}{\hbar}\psi_R.\label{eq:chiral_L}
\end{align}
Writing $\Psi=(\psi_R,\psi_L)^T$, Eqs.~\eqref{eq:chiral_R}--\eqref{eq:chiral_L} become
\begin{equation}
  \boxed{
  i\hbar\partial_t\Psi
  =-i\hbar c\sigma_z\partial_x\Psi
  +mc^2\sigma_x\Psi.}
  \label{eq:dirac_1d}
\end{equation}
This is the $1+1$ dimensional Dirac equation.

Massless modes have $\theta=0$ and remain pure directed tangency streams. Massive modes repeatedly reverse, producing the fractal zigzag path that appears as quantum behavior after coarse-graining.
